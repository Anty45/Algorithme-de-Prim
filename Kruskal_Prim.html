Eleve : Anty KOFFI¶

Introduction¶

Suivant qu'on soit dans l'algorithme de Prim ou de Kruskal ces implmentations different. En effet, les structures de donnes du type dictionnaire permettent de modeliser plus facilement le problme de Prim. Malheuresement, il en resulte que plus la taille du graphe augmente, plus il est difficile d'aller chercher les donnes "en profondeur". Et donc, plus le temps de calcul augmente.

Tandis que les listes permettent de modeliser plus facilement le probleme de Kruskal. Les rsultats sont excellents pour des graphes allant jusqu' 20 30 milles points

def euclidean_distance(coordonne):
    """
    :param coordonne: l'ensemble des coordonnees des points a notre disposition

    :return: Distance euclidienne entre les sommets du graphe

    Le graphe etant Hamiltonien, tout les sommets sont adjacents deux  deux. Les sommets du graphe sont reprsents
    par les cls,valeurs(x,y) d'un dictionnaire imbriqu au sommet considr. Ce sommet reprsente la cl du
    dictionnaire.

    """
    k, v, abscice, ordonne = 0, 0, 0, 0
    point = []
    arete = {}
    distance = 0
    for sommet, value in coordonne.items():
        k = sommet
        v = value
        point.append(k)
        arete[k] = {}
        for v1, v2 in v.items():
            x = v1
            y = v2
        for s_prim, b in coordonne.items():
            if s_prim != k:
                for x_prim, y_prim in b.items():
                    distance = (math.sqrt(((x - x_prim) ** 2) + ((y - y_prim) ** 2)))
                if k not in arete:
                    arete[k] = {}
                arete[k].update({s_prim: distance})
                distance = 0
    return arete, point

import numpy as np 
def prim(distance, sommet, x0):
    """
    :param distance: Type dictionnaire. C'est le graphe avec les sommets ainsi que les aretes
    :param sommet: Type liste. liste de touts les sommet
    :param x0: Type int. C'est le sommet permettant de demarrer l'algorithme
    :return: la liste des arcs ainsi que leur valuation et l'arborescence du graphe
    """
    # Initialisation

    T = []
    sommet_T = [x0]
    n = len(sommet)
    pred_marqueur = {}
    pred = {}

    # Iteration k

    for k in range(n - 1):

        min_edge = float('Inf')
        minimums_by_edge = {}
        for x in sommet_T:

            liste_sommet_atteignable = [y for y in list(distance[x].keys()) if y not in sommet_T]
            for y in liste_sommet_atteignable:
                if distance[x][y] < min_edge:
                    min_edge = distance[x][y]
                    S = y
                    pred_marqueur[S] = {x: min_edge}
            minimums_by_edge[S] = min_edge

        next_edge = min(list(minimums_by_edge.values()))
        T.append(next_edge)
        ind_val_arc = np.argmin(list(minimums_by_edge.values()))
        next_sommet = list(minimums_by_edge.keys())[ind_val_arc]
        sommet_T.append(next_sommet)
        pred[next_sommet] = pred_marqueur[next_sommet]
        # print(sommet_T)

    return T, sommet_T, pred

def euclidean_distance_kruskal(coordonne):
    """
    :param coordonne: l'ensemble des coordonnees des points a notre disposition

    arete = []
    distance = 0
    for i in range(0,len(coordonne)):
        point.append(coordonne[i][0])

            distance = 0

    return arete, point

def kruskal(edge_kruskal, sommet):
    # Initialisation

    T = []
    classe_edge = {}
    n = len(sommet)

    # Definition des classes 
    for i in range(n):
        classe_edge[sommet[i]] = sommet[i]
    
    # Trie des aretes
    F = sorted(edge_kruskal, key=lambda x: x[2])

    j = 0
    while len(T) < (n - 1):

        min_edge = F[0]
        
        F.pop(0) #puisque la liste est trie, on ne modifie que la premiere occurence  chaque tour de boucle 
        # Detection de cycle
        c0 = classe_edge[min_edge[0]]
        c1 = classe_edge[min_edge[1]]

        # On fait la mise  jour si les deux sommets n'appartiennent pas  la mme la classe 
        
        if (classe_edge[min_edge[0]] != classe_edge[min_edge[1]]):
            c0 = classe_edge[min_edge[0]]
            c1 = classe_edge[min_edge[1]]
            for s in sommet:
                # fusion des classes
                if classe_edge[s] == c1:
                    classe_edge[s] = c0
            T.append(min_edge)
        j = j + 1
    print(len(T))
    return (T)

def union(vertice1, vertice2,parent,rank):
    root1 = compresseur(vertice1,parent)
    root2 = compresseur(vertice2,parent)
    if root1 != root2:
        if rank[root1] > rank[root2]:
            parent[root2] = root1
        else:
            parent[root1] = root2
        if rank[root1] == rank[root2]:
             rank[root2] += 1


def compresseur(s,parent):  # fonction find pour la compression de chemin
    if parent[s] != s:  # Le seul sommet qui a pour parent lui mme est la racine
        parent[s] = compresseur(parent[s],parent)  # On cre une fonction recursive qui va recuperer le parent tant qu'elle n'ateint pas la racine
    return parent[s]


def make_set(s,parent,rank):
    parent[s] = s
    rank[s] = 0


def kruskal_optimiser(edge_kruskal, sommet):
   
    # Initialisation
    parent = {}
    rank = {}

    for s in sommet:
        make_set(s,parent,rank)
    T = []
    classe_edge = {}
    n = len(sommet)

    # Definition des classes
    for i in range(n):
        classe_edge[sommet[i]] = sommet[i]

    # Trie des aretes
    F = sorted(edge_kruskal, key=lambda x: x[2])

    # Compression de chemin
    for element in F:
        s1, s2, weight = element[0], element[1], element[2]
        if compresseur(s1,parent) != compresseur(s2,parent):
            union(s1, s2,parent,rank)
            T.append(element)
